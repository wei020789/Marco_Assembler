COPY	START	0
RDBUFF	MACRO	&INDEV,&BUFADR,&RECLTH
	CLEAR	X
	CLEAR	A
	CLEAR	S
	+LDT	#4096
	TD	=X'&INDEV'
	JEQ	*-3
	RD	=X'&INDEV'
	COMPR	A,S
	JEQ	*+11
	STCH	&BUFADR,X
	TIXR	T
	JLT	*-19
	STX	&RECLTH
	MEND

FIRST	STL	RETADR
CLOOP	RDBUFF	F1,BUFFER,LENGTH
	LDA	LENGTH
	COMP	#0
	JEQ	ENDFIL
	J	CLOOP
ENDFIL	J	@RETADER
EOF	BYTE	C'EOF'
THREE	WORD	3
RETADR	RESW	1
LENGTH	RESW	1
BUFFER	RESB	4096
	END	FIRST
